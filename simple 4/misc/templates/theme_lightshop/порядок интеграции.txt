$('.simple-content input[type=radio]').parent().removeClass('checked');

$('.radio label, .checkbox label, .checkout__radio-item ').each(function() {
  var $label = $(this);
  var $input = $label.find('input');
  $input.insertBefore($label);
  $label.prepend($('<span/>'));

  if (!$input.attr('id')) {
    $input.attr('id', $input.attr('name'));
  }
  
  $label.attr('for', $input.attr('id'));
});

$('.checkout__radio-item img+label').each(function() {
  var $label = $(this);
  var $parent = $label.parent();
  var $img = $parent.find('img');
  var $input = $parent.find('input');
  
  $img.insertBefore($input).wrap('<span class="checkout__input-box-title"/>');
});
